import tkinter as tk

def celsius_to_fahrenheit():
    """Конвертация Цельсия в Фаренгейты"""
    try:
        c = float(celsius_var.get())
        f = c * 9/5 + 32
        fahrenheit_var.set(f"{f:.1f}")
    except ValueError:
        pass

def fahrenheit_to_celsius():
    """Конвертация Фаренгейтов в Цельсии"""
    try:
        f = float(fahrenheit_var.get())
        c = (f - 32) * 5/9
        celsius_var.set(f"{c:.1f}")
    except ValueError:
        pass

root = tk.Tk()
root.title("Конвертер температур")
root.geometry("400x200")

# Переменные для хранения значений
celsius_var = tk.StringVar()
fahrenheit_var = tk.StringVar()

# Привязываем функции к изменениям переменных
celsius_var.trace("w", lambda *args: celsius_to_fahrenheit())
fahrenheit_var.trace("w", lambda *args: fahrenheit_to_celsius())

# Поле для Цельсия
tk.Label(root, text="Цельсий:").pack(pady=5)
celsius_entry = tk.Entry(root, textvariable=celsius_var, font=("Arial", 14))
celsius_entry.pack(pady=5)

# Поле для Фаренгейта
tk.Label(root, text="Фаренгейт:").pack(pady=5)
fahrenheit_entry = tk.Entry(root, textvariable=fahrenheit_var, font=("Arial", 14))
fahrenheit_entry.pack(pady=5)

root.mainloop()
