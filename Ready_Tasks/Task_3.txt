import tkinter as tk
from tkinter import messagebox

def add_task():
    """Добавление задачи в список"""
    task = task_entry.get().strip()
    if task:
        task_listbox.insert(tk.END, task)
        task_entry.delete(0, tk.END)
        update_add_button_state()

def delete_task():
    """Удаление выбранной задачи"""
    try:
        selected_index = task_listbox.curselection()[0]
        task_listbox.delete(selected_index)
    except IndexError:
        messagebox.showwarning("Внимание", "Выберите задачу для удаления")

def on_double_click(event):
    """Обработка двойного клика для удаления"""
    delete_task()

def update_add_button_state():
    """Обновление состояния кнопки Добавить"""
    if task_entry.get().strip():
        add_button.config(state=tk.NORMAL)
    else:
        add_button.config(state=tk.DISABLED)

root = tk.Tk()
root.title("Список дел")
root.geometry("400x400")

# Поле ввода
task_entry = tk.Entry(root, font=("Arial", 14))
task_entry.pack(pady=10, padx=20, fill=tk.X)
task_entry.bind("<KeyRelease>", lambda e: update_add_button_state())

# Фрейм для кнопок
button_frame = tk.Frame(root)
button_frame.pack(pady=10)

add_button = tk.Button(button_frame, text="Добавить", command=add_task, 
                      state=tk.DISABLED)
add_button.pack(side=tk.LEFT, padx=5)

delete_button = tk.Button(button_frame, text="Удалить", command=delete_task)
delete_button.pack(side=tk.LEFT, padx=5)

# Список задач
task_listbox = tk.Listbox(root, font=("Arial", 12), height=15)
task_listbox.pack(pady=10, padx=20, fill=tk.BOTH, expand=True)
task_listbox.bind("<Double-Button-1>", on_double_click)

# Полоса прокрутки
scrollbar = tk.Scrollbar(task_listbox)
scrollbar.pack(side=tk.RIGHT, fill=tk.Y)
task_listbox.config(yscrollcommand=scrollbar.set)
scrollbar.config(command=task_listbox.yview)

root.mainloop()
